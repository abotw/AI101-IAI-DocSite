
```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
------------------------------------------------------------
Dofbot Arm Control Experiment 6.6 — Up, Down, Left, and Right Swing
------------------------------------------------------------
Author: Yahboom
File Path: /home/yahboom/Dofbot/3.ctrl_Arm/6.left_right.ipynb
Description:
    This experiment demonstrates how to make the robotic arm
    swing up, down, left, and right, and then return to its
    upright (centered) position.

Experiment Idea:
    - Control servo #3 and servo #4 simultaneously to achieve
      up-and-down movement of the arm.
    - Control servo #1 to swing the arm left and right.
    - Finally, reset all servos to the neutral 90° position.
------------------------------------------------------------
"""

import time
from Arm_Lib import Arm_Device


def main():
    """
    Main control function for the Dofbot arm.

    The function performs a continuous loop:
        1. Reset all servos to the middle position (90°).
        2. Move servos #3 and #4 for up/down motion.
        3. Move servo #1 for left/right swinging.
        4. Return the arm to the upright position.
    """

    # Create an instance of the Arm_Device class
    arm = Arm_Device()
    time.sleep(0.1)

    # Step 1: Reset all servos to their neutral positions
    arm.Arm_serial_servo_write6(90, 90, 90, 90, 90, 90, 500)
    time.sleep(1)

    try:
        while True:
            # Step 2: Move servos #3 and #4 to swing up/down
            # Servo #3 -> 0°, Servo #4 -> 180°
            arm.Arm_serial_servo_write(3, 0, 1000)
            time.sleep(0.001)
            arm.Arm_serial_servo_write(4, 180, 1000)
            time.sleep(1)

            # Step 3: Move servo #1 to swing left/right
            # Servo #1 -> right (180°)
            arm.Arm_serial_servo_write(1, 180, 500)
            time.sleep(0.5)

            # Servo #1 -> left (0°)
            arm.Arm_serial_servo_write(1, 0, 1000)
            time.sleep(1)

            # Step 4: Reset all servos to neutral position again
            arm.Arm_serial_servo_write6(90, 90, 90, 90, 90, 90, 1000)
            time.sleep(1.5)

    except KeyboardInterrupt:
        print("\nProgram closed by user.")
    finally:
        # Always release the arm object when exiting
        del arm
        print("Arm object released.")


if __name__ == "__main__":
    main()
```
